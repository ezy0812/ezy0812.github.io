vector

#include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> list;
    int n;

    cout << "Enter the number of the items in the vector: " << endl;
    cin >> n;

    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;
        list.push_back(a);
    }

    cout<<"The length of the vector called list is " << list.size() << endl;
    //迭代器遍历
    vector<int>::iterator p1;
    cout << "The value of the vector is: " << endl;
    for (p1 = list.begin(); p1 != list.end(); p1++)
        cout << *p1 << endl;

    list.size();//元素个数
    list.clear();//清空
    list.empty();//是否为空
    list.erase(p1);//删除某个迭代器所在位置的数字
    list.pop_back();//删除最后一个元素

    return 0;
}

algorithm

#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;
int cmp(int a, int b) {
    return a > b;
}
int main() {
    int arr[] = { 2,3,1,5,4 };
    sort(arr, arr + 5);//快速排序，O(nlogn) 

    vector<int> arr2{ 2,3,1,5,4 };
    sort(arr2.begin(), arr2.end());

    sort(arr, arr + 5, cmp);//自定义比较函数

    //最大值，最小值
    min(1, 2); max(1, 2);  //O(1)
    min_element(arr2.begin(), arr2.end());//O(n)
    max_element(arr2.begin(), arr2.end());//O(n)

    //数组未排好序，将数组中第n小的数放到第n个位置，并且保证它左边的都比它小，右边的都比它大
    int n = 3;
    nth_element(arr2.begin(), arr2.begin() + n, arr2.end());//O(n)

    swap(arr[0], arr[1]);//O(1)

    reverse(arr2.begin(), arr2.end());//反转数组，O(n)

    //数组已排好序，对其进行去重操作，返回去重后的结束指针
    vector<int> arr3 { 1,1,2,2,3,3 };
    int newlength=unique(arr3.begin(), arr3.end())-arr3.begin();

    //查找元素是否存在
    bool isExist = binary_search(arr2.begin(), arr2.end(), 1);//O(logn)

    //将一个数插入一个有序数组，应该插入到哪个位置
    //lower_bound返回第一个插入位置的指针，upper_bound返回最后一个位置的指针 
    //O(logn)
    vector<int> arr4{ 1,2,3,4,5 };
    int firstLocation = lower_bound(arr4.begin(), arr4.end(), 2) - arr4.begin();
    int lastLocation=upper_bound(arr4.begin(), arr4.end(), 2) - arr4.begin();

    return 0;
}

#include <iostream>
#include <bitset>

using namespace std;

int main() {
    bitset<1000> bst;
    bst[0] = 1; bst.set(0);
    bst[0] = 0; bst.reset(0);
    bst << 1; bst >> 1; bst ^= 1; bst &= 1;
    bst.count();//1的个数
    bst.to_string();
    bst.to_ullong();

    return 0;
}
