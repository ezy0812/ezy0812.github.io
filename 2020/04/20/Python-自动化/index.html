os模块

路径中的斜杠(/)与反斜杠()

windows中采用\，Mac和Linux采用/
\在Python中用于转义，所以变成了两个\

>>> os.getcwd()
'F:\\blog\\hexo'

Python可以自动处理路径连接

>>> os.path.join('myproject','AI') 
'myproject\\AI'

而在Linux环境下会输出'myproject/AI'

搜索、匹配文件名称

字符串内置方法

>>> 'abc.txt'.startswith('ab')
True
>>> 'abc.txt'.endswith('xt')

使用glob模块

>>> import glob
>>> glob.glob('*.json')
['db.json', 'package-lock.json', 'package.json']
>>> glob.glob('pack*.json') 
['package-lock.json', 'package.json']
模式 意义
* 匹配所有
? 匹配任意单个字符
[seq] 匹配seq中的任何字符
[!seq] 匹配任何不再seq中的字符

写入，读取文件

>>> f=open('.gitignore','r',encoding='utf-8')  
>>> text=f.readlines()
>>> print(text)
['.DS_Store\n', 'Thumbs.db\n', 'db.json\n', '*.log\n', 'node_modules/\n', 'public/\n', '.deploy*/']
>>> f.close()

一种建议的写法，不管程序是否正常运行都会关闭文件，不需要再单独写close()

>>> with open('.gitignore','r',encoding='utf-8') as f:
...     text=f.readlines()
...     print(text)
... 
['.DS_Store\n', 'Thumbs.db\n', 'db.json\n', '*.log\n', 'node_modules/\n', 'public/\n', '.deploy*/']

写文件

>>> with open('1.txt','w',encoding='utf-8') as f:      
...     text='first line\nsecondline\n'
...     f.write(text)
  • ‘w’表示写入，若文件不存在则会创建新文件，若存在，则会将文件原内容清空，然后写入内容
  • ‘a’表示追加，若件不存在则会创建新文件，若存在，则会在文件最后追加内容，而不清空原内容

创建临时文件及文件夹

from tempfile import TemporaryFile

with TemporaryFile('w+') as f:
    f.write('Hello')
    f.seek(0)
    data=f.readlines()
    print(data)
  • ‘w+’表示写入及读取文件
  • seek(0)表示回到文件开头位置
  • 在程序运行完之后会自动删掉临时文件 
from tempfile import TemporaryDirectory

with TemporaryDirectory() as tmp_folder:
    print(f'临时文件夹已创建：{tmp_folder}')

复制、移动、重命名

使用shutil模块

复制文件有两种使用方式

  1. 第二个参数写某个文件夹位置，则复制到该文件夹下
  2. 第二个参数写某个文件路径，复制到这个路径并且重命名
>>> import shutil
>>> shutil.copy('mysql.txt','./sqlmap')
'./sqlmap\\mysql.txt'
>>> shutil.copy('mysql.txt','./sqlmap/2.txt') 
'./sqlmap/2.txt'

复制文件夹

>>> shutil.copytree('GAME','game2')
'game2'

移动文件或文件夹

>>> shutil.move('mysql.txt','GAME/')
'GAME/mysql.txt'
>>> shutil.move('mysql.txt','GAME/2.txt') 
'GAME/2.txt'
>>> shutil.move('GAME','sqlmap/')
'sqlmap/GAME'

记住第二个参数若为文件夹最后要加斜杠/

重命名文件或文件夹

>>> os.rename('mysql.txt','database_notes.txt')
>>> os.rename('GAME','LOL')

删除文件

只能删文件，如果给了一个文件夹路径则会报错

>>> os.remove('remove.txt')

删除文件夹

>>> shutil.rmtree('remove')

创建和解压压缩包

读取压缩包

>>> import zipfile
>>> with zipfile.ZipFile('yasuo.zip','r') as zipobj:
...     print(zipobj.namelist())
... 
['1.txt', '2.txt', '3.txt']

若文件列表中有中文文件名，则需要调整编码

>>> with zipfile.ZipFile('yasuo.zip','r') as zipobj:
...     for filename in zipobj.namelist():
...             print(filename.encode('cp437').decode('gbk'))
... 
1.txt
2.txt
3.txt
中文.txt
>>> with zipfile.ZipFile('yasuo.zip','r') as zipobj:
...     for filename in zipobj.namelist():
...             info=zipobj.getinfo(filename)
...             new_filename=filename.encode('cp437').decode('gbk')
...             print(new_filename,info)
... 
1.txt <ZipInfo filename='1.txt' external_attr=0x20 file_size=0>

3.txt <ZipInfo filename='3.txt' external_attr=0x20 file_size=0>
中文.txt <ZipInfo filename='╓╨╬─.txt' external_attr=0x20 file_size=0>

info属性：

  • .file_size 原始文件大小
  • .compress_size 压缩后大小
    单位为字节

解压压缩包

import zipfile

with zipfile.ZipFile('yasuo.zip','r') as zipobj:
    zipobj.extract('1.txt')

解压所有文件，可选参数为path

import zipfile

with zipfile.ZipFile('yasuo.zip','r') as zipobj:
    zipobj.extractall(path='./extract/')

若压缩包有密码，则使用可选参数pwd

import zipfile

with zipfile.ZipFile('yasuo.zip','r') as zipobj:
    zipobj.extractall(path='./extract/',pwd=b'123')

不要忘记pwd参数加b

创建压缩包

import zipfile

file_list=['yasuo.zip','extract/']
with zipfile.ZipFile('new.zip','w') as zipobj:
    for file in file_list:
        zipobj.write(file)

注意with一行是w,即写入模式

向已有压缩包添加文件

import zipfile

with zipfile.ZipFile('new.zip','a') as zipobj:
    zipobj.write('sqli.txt')

注意with一行是a,即追加模式

处理excel文件

Excel表格术语

  • 行：row,从1开始计数
  • 列：column
  • 格子：cell
  • 表：sheet

打开及读取表格数据

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
print(workbook.sheetnames)
sheet=workbook['Sheet1']
print(sheet.dimensions)         #.dimensions是从左上角坐标到右下角坐标

获取一系列格子

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active                       #.active 打开活跃的/唯一的表格
cell=sheet['A1']
print(cell.value)
print(cell.row,cell.column,cell.coordinate)
cell=sheet.cell(row=1,column=1)             #根据坐标获取具体cell
print(cell.row,cell.column,cell.coordinate)
cells=sheet['A1:A3']                        #获取一系列格子
print(cells)
cells=sheet['A']                            #获取A列数据
print(cells)
cells=sheet['A:C']                          #获取A到C列数据
print(cells)
cells=sheet[2]                              #获取第2行数据
print(cells)
cells=sheet['2:3']                          #获取第2到3行数据
print(cells)
for row in sheet.iter_rows(min_row=1,max_row=2,min_col=2,max_col=3):    #指定行和列的范围，按行获取
    print(row)
for col in sheet.iter_cols(min_row=1,max_row=2,min_col=2,max_col=3):    #指定行和列的范围，按列获取
    print(col)
for row in sheet.rows:                      #所有行
    print(row)

向表格中写入内容

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active

workbook.save(filename='example.xlsx')

插入一行

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
data=[
    ['a',1],
    ['b',2],
    ['c',3],
    ['d',4],
]
for row in data:
    sheet.append(row)
workbook.save(filename='example.xlsx')

插入公式

from openpyxl import load_workbook
from openpyxl.utils import FORMULAE

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
sheet['B7']='=AVERAGE(B1:B6)'
workbook.save(filename='example.xlsx')

print(FORMULAE)                 #查看openpyxl支持的公式

插入和删除一列、多列、一行、多行

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
sheet.insert_cols(idx=2)                #在idx列左边插入一列空的列
sheet.insert_cols(idx=4,amount=2)       #在idx列左边插入2列空的列
sheet.insert_rows(idx=2)                #在idx行上边插入一行空的行
sheet.insert_rows(idx=2,amount=2)       #在idx行上边插入2行空的行
sheet.delete_cols(idx=2)                #删除idx列    
sheet.delete_cols(idx=2,amount=2)       #删除从idx列开始，向右共2列
sheet.delete_rows(idx=2)                #删除idx行    
sheet.delete_rows(idx=2,amount=2)       #删除从idx行开始，向下共2行
workbook.save(filename='example.xlsx')

移动格子

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
sheet.move_range("A1:B2",rows=2,cols=2)        #正整数为向下或向右，负整数为向左或向上
workbook.save(filename='example.xlsx')

创建、删除、复制、改名sheet

from openpyxl import load_workbook

workbook=load_workbook(filename='example.xlsx')
print(workbook.sheetnames)
workbook.create_sheet('sheet2')         #创建
print(workbook.sheetnames)

sheet=workbook['sheet2']
workbook.remove(sheet)                  #删除

sheet=workbook['Sheet1']
workbook.copy_worksheet(sheet)          #复制

关于样式

修改字体样式

from openpyxl import load_workbook
from openpyxl.styles import Font

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
cell=sheet['A2']
font=Font(name='华文琥珀',size=12,bold=True,italic=True,color='FF0000')
cell.font=font
print(font.name,font.size,font.bold,font.italic)                           #获取字体信息
workbook.save(filename='example.xlsx')

设置对齐样式

from openpyxl import load_workbook
from openpyxl.styles import Alignment

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
cell=sheet['A3']
alignment=Alignment(horizontal='center',vertical='center',text_rotation=45,wrap_text=True)
#horizontal=水平对齐模式 vertical=垂直对齐模式 text_rotation=旋转角度 wrap_text=是否自动换行
cell.alignment=alignment
workbook.save(filename='example.xlsx')

水平对齐:’distributed’,’justify’,’center’,’left’,’fill’,’centerContinuous’,’right’,’general’
垂直对齐:’bottom’,’distributed’,’justify’,’center’,’top’

设置边框样式

from openpyxl import load_workbook
from openpyxl.styles import Side,Border

workbook=load_workbook(filename='example.xlsx')
sheet=workbook.active
cell=sheet['A3']
side1=Side(style='thin',color='FF000000')       #style=边线样式,color=边线颜色
side2=Side(style='dotted',color='FFFF0000')  
border=Border(left=side1,right=side2,top=side2,bottom=side1)
cell.border=border
workbook.save(filename='example.xlsx')
