作为红队，首先要做到隐藏自身，防止被溯源

不需要实名的VPS

vultr

可以使用CentOS SELinux 7 x64

[root@vultr ~]# yum install net-tools

# 安装msf
[root@vultr ~]# curl https://raw.githubusercontent.com/rapid7/metasploit-omnibus/master/config/templates/metasploit-framework-wrappers/msfupdate.erb > msfinstall
[root@vultr ~]# chmod +x msfinstall 
[root@vultr ~]# ./msfinstall 
[root@vultr ~]# rm -rf msfinstall

[root@vultr ~]# yum install nc -y
[root@vultr ~]# yum install nmap -y

# sqlmap
[root@vultr ~]# wget https://codeload.github.com/sqlmapproject/sqlmap/legacy.tar.gz/master
[root@vultr ~]# tar -zxvf master
[root@vultr ~]# mv sqlmapproject-sqlmap-97b88b0 sqlmap
[root@vultr ~]# rm -rf master

# 安装java
[root@vultr ~]# wget https://repo.huaweicloud.com/java/jdk/8u171-b11/jdk-8u171-linux-x64.tar.gz
[root@vultr ~]# tar -zxvf jdk-8u171-linux-x64.tar.gz -C /usr/lib/
[root@vultr ~]# echo "export JAVA_HOME=/usr/lib/jdk1.8.0_171/" >> /etc/profile
[root@vultr ~]# echo "export JRE_HOME=/usr/lib/jdk1.8.0_171/jre" >> /etc/profile
[root@vultr ~]# echo "export PATH=\$JAVA_HOME/bin:\$JAVA_HOME/jre/bin:\$PATH" >> /etc/profile
[root@vultr ~]# echo "export CLASSPATH=\$CLASSPATH:.:\$JAVA_HOME/lib:\$JAVA_HOME/jre/lib" >> /etc/profile
[root@vultr ~]# source /etc/profile
[root@vultr ~]# java -version
java version "1.8.0_171"
Java(TM) SE Runtime Environment (build 1.8.0_171-b11)
Java HotSpot(TM) 64-Bit Server VM (build 25.171-b11, mixed mode)

# 安装CS （客户端需要java版本1.8，高版本不行）
# 先安装一个传输工具，上传cobaltstrike4.3.zip
# 运行前先关闭防火墙
[root@vultr cobaltstrike4.3]# systemctl stop firewalld.service
[root@vultr cobaltstrike4.3]# systemctl disable firewalld.service
[root@vultr ~]# yum -y install lrzsz
[root@vultr ~]# rz -y
[root@vultr ~]# yum install unzip
[root@vultr ~]# unzip cobaltstrike4.3.zip 
[root@vultr cobaltstrike4.3]# chmod +x teamserver
[root@vultr cobaltstrike4.3]# ./teamserver ip password

钓鱼域名

https://my.freenom.com/clientarea.php?action=domains

注册一个域名

该网站的my details里面地区设为vpn的地区

注册CDN

https://www.cloudflare.com/zh-cn/

在账号中添加注册的域名，然后添加A记录，解析到VPS的IP地址

去域名管理商里修改Nameservers为cloudflare的域名即可

去缓存的配置界面开启CDN的开发模式（3小时自动失效，还需要自己手动开启）+ Always Online™Beta

SSL/TLS 加密模式要为灵活

设置完之后去全国ping里面ping一下是否有CDN

渗透使用全英文的操作系统

上线端口

cloudflare可用的上线端口：

  • 对于http 80 8080 8880 2052 2082

  • 对于https 443 2053 2083 2096

不要用reverse_tcp上线，否则会暴露服务器真实IP，使用http和https上线才是CDN的IP
