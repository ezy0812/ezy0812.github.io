some configuration before spider

将settings.py中的ROBOTSTXT_OBEY设置为False，即不遵循robots协议

# Obey robots.txt rules
ROBOTSTXT_OBEY = False

设置UA

DEFAULT_REQUEST_HEADERS = {
  'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',

  'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/96.0.4655.6 Safari/537.36'
}

demo.py

import scrapy
from qsbk.items import QsbkItem


class DemoSpider(scrapy.Spider):
    name = 'demo'
    allowed_domains = ['qiushibaike.com']
    start_urls = ['https://www.qiushibaike.com/text/page/1/']

    def parse(self, response):
        divs = response.xpath('//*[@id="content"]/div/div[2]/div')
        for each in divs:
            author = each.xpath('./div[1]/a[2]/h2/text()').get().strip()
            print(author)
            content = each.xpath('./a[1]/div/span//text()').getall()
            content = "".join(content).strip()
            item = QsbkItem(author=author, content=content)
            # 将数据移交给pipelines处理
            yield item

        next_url = response.xpath('//ul[@class="pagination"]/li[last()]/a/@href').get()
        if not next_url:
            return
        else:
            yield scrapy.Request("https://www.qiushibaike.com/"+next_url, callback=self.parse)

# Define your item pipelines here
#
# Don't forget to add your pipeline to the ITEM_PIPELINES setting

# useful for handling different item types with a single interface
from itemadapter import ItemAdapter

from scrapy.exporters import JsonLinesItemExporter


class QsbkPipeline:
    def __init__(self):
        self.fp = open("duanzi.json", "wb")
        self.exporter = JsonLinesItemExporter(self.fp, ensure_ascii=False, encoding='utf-8')

    # 启动爬虫时调用的函数
    def open_spider(self, spider):
        pass

    # 当有item传过来时被调用
    def process_item(self, item, spider):
        self.exporter.export_item(item)
        return item

    # 爬虫完成时调用的函数
    def close_spider(self, spider):
        self.fp.close()

想要pipelines运行，需要在settings中ITEM_PIPELINES取消注释

ITEM_PIPELINES = {
   'qsbk.pipelines.QsbkPipeline': 300,
}

300指优先级数字，可以自己修改，因为该项目中只有一个爬虫，所以不需要修改

DOWNLOAD_DELAY可以设置爬虫延时，防止爬取过快

DOWNLOAD_DELAY = 1

CrawlSpider

继承自spider，增加了新的功能，可以定义爬取的url的规则，使scrapy碰到满足条件的url都进行爬取，而不需要手动yield Request

scrapy startproject wxapp
cd wxapp
scrapy genspider -t crawl wxapp_spider "wxapp-union.com"

wxapp_spider.py

import scrapy
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from wxapp.items import WxappItem


class WxappSpiderSpider(CrawlSpider):
    name = 'wxapp_spider'
    allowed_domains = ['wxapp-union.com']
    start_urls = ['https://www.wxapp-union.com/portal.php?mod=list&catid=2&page=1']

    rules = (
        Rule(LinkExtractor(allow=r'.+mod=list&catid=2&page=\d'), follow=True),
        Rule(LinkExtractor(allow=r'article-.+\.html'), callback="parse_detail", follow=False),
    )

    def parse_detail(self, response):
        title = response.xpath('//*[@id="ct"]/div[1]/div/div[1]/div/div[2]/div[1]/h1/text()').get()
        p_label = response.xpath('//p[@class="authors"]')
        author = p_label.xpath('./a/text()').get()
        pub_time = p_label.xpath('./span/text()').get()
        content = response.xpath('//td[@id="article_content"]//text()').getall()
        content = "".join(content).strip()
        item = WxappItem(title=title, author=author, pub_time=pub_time, content=content)
        yield item

LinkExtractor链接提取器中注意allow（想要爬取的url规则），follow（在当前页面中碰到同样规则的url是否需要继续跟进爬取），callback（对于爬取到的页面要执行的回调函数）
