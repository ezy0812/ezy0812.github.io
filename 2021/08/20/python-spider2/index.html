re 模块

.* 贪婪匹配

.*? 懒惰匹配

s = '玩儿吃鸡游戏，晚上一起玩游戏，干嘛呢？打游戏啊'
r1 = re.compile(r'玩儿.*游戏', re.S)
r2 = re.compile(r'玩儿.*?游戏', re.S)
result1 = r1.findall(s)
print(result1)
result2 = r2.findall(s)
print(result2)

output

['玩儿吃鸡游戏，晚上一起玩游戏，干嘛呢？打游戏']
['玩儿吃鸡游戏']

Process finished with exit code 0

贪婪匹配会尽可能多的匹配内容，而懒惰匹配则相反

关于re.S
a = """sdfkhellolsdlfsdfiooefo:
877898989worldafdsf"""
b = re.findall('hello(.*?)world',a)
c = re.findall('hello(.*?)world',a,re.S)
print ('b is ' , b)
print ('c is ' , c)

output

b is  []
c is  ['lsdlfsdfiooefo:\n877898989']

Process finished with exit code 0

如果不使用re.S参数，则只在每一行内进行匹配，如果一行没有，就换下一行重新开始。

而使用re.S参数以后，正则表达式会将这个字符串作为一个整体，在整体中进行匹配。

以变量形式取出自己想要的内容
s = """
<div class='jay'><span id='1'>jay</span></div>
<div class='jj'><span id='2'>jj</span></div>
<div class='joli'><span id='3'>joli</span></div>
<div class='tom'><span id='4'>tom</span></div>
"""

r1 = re.compile(r"<div class='.*?'><span id='\d+'>(?P<name>.*?)</span></div>", re.S)

result = r1.finditer(s)
for i in result:
    print(i.group("name"))

output

jay
jj
joli
tom

Process finished with exit code 0

import re
import requests
import time

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                  "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/94.0.4606.12 Safari/537.36"
}


def requ(para):
    resp = requests.get(para, headers=headers)
    content = resp.text
    recompile = re.compile(r'<img width="100".*?<span class="title">(?P<name>.*?)</span>'
                           r'.*?<br>(?P<year>.*?)&nbsp.*?property="v:average">(?P<sore>.*?)</span>', re.S)
    res = recompile.finditer(content)
    for i in res:
        print(i.group("name") + "->" + i.group("year").strip() + " -- " + i.group("sore"))
    resp.close()


if __name__ == '__main__':

    for page in range(10):
        url = f'https://movie.douban.com/top250?start={page * 25}&filter='
        requ(url)
        time.sleep(0.5)

注意每次要延时0.5s，以防请求过快

Xpath解析HTML + 抓取猪八戒官网

from lxml import etree

xml = """
<book>
    <id>1</id>
    <name>野花遍地香</name>
    <price>1.23</price>
    <author>
        <nick id="100">jay</nick>
        <nick id="101">tom</nick>
        <nick class="jolin">jolin</nick>
        <nick class="alice">alice</nick>
        <div>
            <nick>aa</nick>
        </div>
        <div>
            <nick>bb</nick>
            <div>
            <nick>cc</nick>
            </div>
        </div>
        <span>
            <nick>ss</nick>            
        </span>
    </author>
</book>
"""

tree = etree.XML(xml)
result = tree.xpath("/book/name/text()")
result = tree.xpath("/book/author/nick/text()")  # ['jay', 'tom', 'jolin', 'alice']
result = tree.xpath("/book/author//nick/text()")  # ['jay', 'tom', 'jolin', 'alice', 'aa', 'bb', 'cc']
result = tree.xpath("/book/author/*/nick/text()")  # ['aa', 'bb', 'ss']
# '//'代表所有nick节点
from lxml import etree
import requests

url = "https://beijing.zbj.com/search/f/?type=new&kw=saas"
headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                  "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/94.0.4606.12 Safari/537.36"
}
resp = requests.get(url, headers=headers)
html = etree.HTML(resp.text)
divs = html.xpath("/html/body/div[6]/div/div/div[2]/div[4]/div[1]/div")
for div in divs:
    try:
        price = div.xpath("./div/div/a[2]/div[2]/div[1]/span[1]/text()")[0].strip("¥")
        name = "saas".join(div.xpath("./div/div/a[2]/div[2]/div[2]/p/text()"))
        company = div.xpath("./div/div/a[1]/div[1]/p/text()")[1].strip()
        address = div.xpath("./div/div/a[1]/div[1]/div/span/text()")[0]
        print(price+" -- "+name+" -- "+company+" -- "+address)
        
    except Exception as e:
        pass

resp.close()

防盗链 - 抓取梨视频

梨视频的视频链接是在XHR中抓取到（https://video.pearvideo.com/mp4/adshort/20210820/1629541188301-15751458_adpkg-ad_hd.mp4）

并且被二次加工为

（https://video.pearvideo.com/mp4/adshort/20210820/cont-1739559-15751458_adpkg-ad_hd.mp4）

即1629541188301 被加工成cont-1739559

使用requests请求之后发现存在防盗链，需要在请求头里加入Referer字段

import requests

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                  "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/94.0.4606.12 Safari/537.36",
    "Referer": "https://pearvideo.com/video_1739559"
}

url = "https://pearvideo.com/video_1739559"
contId = url.split("_")[1]
video_status = f"https://pearvideo.com/videoStatus.jsp?contId={contId}&mrd=0.2949276285579938"
resp = requests.get(video_status, headers=headers)
js = resp.json()
systemTime = js['systemTime']
srcUrl = js['videoInfo']['videos']['srcUrl']
srcUrl = srcUrl.replace(systemTime, f"cont-{contId}")

with open("a.mp4", "wb") as f:
    f.write(requests.get(srcUrl).content)
resp.close()

线程池爬取新发地

from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor
import requests

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                  "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/94.0.4606.12 Safari/537.36"
}


def down_one_page(url, dat):
    resp = requests.post(url, headers=headers, data=dat)
    page_data = resp.json()['list']
    for i in page_data:
        prodName = i['prodName']
        lowPrice = i['lowPrice']
        highPrice = i['highPrice']
        place = i['place']
        unitInfo = i['unitInfo']
        specInfo = i['specInfo']
        pubDate = i['pubDate']
        write_file = prodName+" -- "+lowPrice+" -- "+highPrice+" -- "+place+" -- "+unitInfo+" -- "+specInfo+" -- "+pubDate+"\n"
        with open("xinfadi.txt", "a+", encoding='utf-8') as f:
            f.write(write_file)
    resp.close()


if __name__ == '__main__':
    for i in range(1, 10):
        data = {
            "limit": "20",
            "current": f"{i}"
        }
    with ThreadPoolExecutor(50) as t:
        for i in range(1, 200):
            data = {
                "limit": "20",
                "current": f"{i}"
            }
            t.submit(down_one_page, "http://www.xinfadi.com.cn/getPriceData.html", data)

aiohttp异步爬取整本小说

import asyncio
import aiohttp
import requests
import json
import aiofiles

# https://dushu.baidu.com/api/pc/getCatalog?data={"book_id":"4306063500"}
# https://dushu.baidu.com/api/pc/getChapterContent?data={"book_id":"4306063500","cid":"4306063500|11348579","need_bookinfo":1}


async def down(cid, bid, title):
    data = {
        "book_id": bid,
        "cid": f"{bid}|{cid}",
        "need_bookinfo": 1
    }
    data = json.dumps(data)
    url = f"https://dushu.baidu.com/api/pc/getChapterContent?data={data}"
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as resp:
            dic = await resp.json()
            async with aiofiles.open(file=f'novel/{title}.txt', mode="w", encoding='utf-8') as f:
                await f.write(dic['data']['novel']['content'])


async def getCatalog(url):
    resp = requests.get(url)
    dic = resp.json()
    tasks = []
    for i in dic['data']['novel']['items']:
        title = i['title']
        cid = i['cid']
        tasks.append(asyncio.create_task(down(cid, bookid, title)))

    await asyncio.wait(tasks)
    resp.close()


if __name__ == '__main__':
    bookid = "4306063500"
    url = 'https://dushu.baidu.com/api/pc/getCatalog?data={"book_id":"'+bookid+'"}'
    asyncio.get_event_loop().run_until_complete(getCatalog(url))

页面等待

隐式等待 - 调用implicitly_wait，在获取不可用的元素之前，会先等待一些时间

web.implicitly_wait(10)
web.get("https://www.baidu.com/")

显式等待 - 表明某个条件成立后才执行获取元素的操作，也可以在等待中指定一个最大的时间，如果超过这个时间，就会抛出异常。使用selenium.webdriver.support.ui.WebDriverWait 和 selenium.webdriver.support.expected_conditions配合完成

from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions
try:
    element = WebDriverWait(web, 10).until(expected_conditions.presence_of_element_located((By.ID, "myelement")))
finally:
    web.quit()

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions

driver = webdriver.Chrome()
url = "https://talent.baidu.com/external/baidu/campus.html#/jobList"
driver.get(url)

try:
    div = WebDriverWait(driver, 10).until(
        expected_conditions.presence_of_element_located(
            (By.XPATH, '//*[@id="campus-job-list"]/div/div[2]/div[1]/div/div[2]'))
    )
    list_row = div.find_elements_by_class_name('list-row')
    for each in list_row:
        a_element = each.find_element_by_xpath('./div[1]/a')
        job_link = a_element.get_property('href')
        job_name = a_element.text
        print(job_link+" - "+job_name)
finally:
    driver.quit()
